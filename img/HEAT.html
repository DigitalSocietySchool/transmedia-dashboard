This example shows a simple static heatmap. In this context static means that the data will not be changed at any time after the heatmap creation process.
Click on the heatmap in order to refresh the dataset
  1. <!DOCTYPE html>
  2. <html lang="en">
  3. ...
  4. <div id="heatmapArea" />
  5. <script type="text/javascript" src="heatmap.js"></script>
  6. <script type="text/javascript">
  7. window.onload = function(){
  8.  
  9.     // heatmap configuration
  10.     var config = {
  11.         element: document.getElementById("heatmapArea"),
  12.         radius: 30,
  13.         opacity: 50
  14.     };
  15.     
  16.     //creates and initializes the heatmap
  17.     var heatmap = h337.create(config);
  18.  
  19.     // let's get some data
  20.     var data = {
  21.         max: 20,
  22.         data: [
  23.             { x: 10, y: 20, count: 18 },
  24.             { x: 25, y: 25, count: 14 },
  25.             { x: 50, y: 30, count: 20 }
  26.             // ...
  27.         ]
  28.     };
  29.  
  30.     heatmap.store.setDataSet(data);
  31. };
  32. </script>
  33. </html>
